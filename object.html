Object in Js include property and method

there 3 way to contruct a object in Js

  • 1: Oject Literal 
    						var Person = {
							'name': 'datfeed',
							'age': 8,
							showInfo: function(){
								console.log($this.name + ' ' + $this.age);
							}
						}
  • 2: Contructor pattern 
    						function Person(name,age){
							this.name = name;
							this.age = age;
							this.showInfo = function(){
								console.log($this.name + ' ' + this.age);
							}
						}
						var ps1 = new Person('datfeed',25);
						ps1.showInfo();
  • 3: Using prototype 
    						function Person(){}
						Person.prototype.name = 'datfeed';
						Person.prototype.age = 25;
						Person.prototype.showInfo = function(){
							console.log(this.name + '' + this.age);
						}

						var ps1 = new Person();
						ps1.showInfo();
